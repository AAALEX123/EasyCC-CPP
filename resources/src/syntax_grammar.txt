PROG            -> CLASSDECLRPT PROGBODY
CLASSDECLRPT    -> CLASSDECL CLASSDECLRPT | EPSILON
CLASSDECL       -> #startClass# T_CLASS CLASSIDEN #className# CLASSBODY SEMICOLON
CLASSBODY       -> T_OPEN_CURLY VARDECLRPTFUNC CLOSECURLY
VARDECLRPTFUNC  -> TYPE VARFUNCIDEN VARFUNCDISAMB | EPSILON
VARFUNCDISAMB   ->  ARRAYSIZERPT  SEMICOLON VARDECLRPTFUNC | T_OPEN_PAREN FPARAMS CLOSEPAREN FUNCBODY SEMICOLON   FUNCDEFRPT
FUNCDEFRPT      -> TYPE FUNCDEFIDEN OPENPAREN FPARAMS CLOSEPAREN FUNCBODY SEMICOLON   FUNCDEFRPT | EPSILON
ARRAYSIZERPT    -> ARRAYSIZE ARRAYSIZERPT | EPSILON
ARRAYSIZE       -> T_OPEN_SQUARE ARRAYSIZEINT  ARRAYSIZECLOSQ
FPARAMS         -> TYPE  FPARAMSIDEN  ARRAYSIZERPT  FPARAMSTAILRPT | EPSILON
FPARAMSTAIL     -> T_COMMA TYPE  FPARAMSIDEN  ARRAYSIZERPT
FPARAMSTAILRPT  -> FPARAMSTAIL FPARAMSTAILRPT | EPSILON
PROGBODY        -> T_PROGRAM  FUNCBODY SEMICOLON  FUNCDEFRPT
FUNCBODY        -> T_OPEN_CURLY VARDECLRPTST CLOSECURLY
VARDECLRPTST    -> T_INT_TYPE  IDENTIFIER  ARRAYSIZERPT  SEMICOLON VARDECLRPTST
                | T_FLOAT_TYPE  IDENTIFIER  ARRAYSIZERPT  SEMICOLON VARDECLRPTST
                | T_IDENTIFIER VARDECLRPTSTDIS
                | T_IF OPENPAREN EXPR  CLOSEPAREN THEN  STATBLOCK ELSE  STATBLOCK  SEMICOLON STATEMENTRPT
                | T_FOR  OPENPAREN TYPE  IDENTIFIER     ASSIGNOP  EXPR   SEMICOLON RELEXPR   SEMICOLON ASSIGNSTAT  CLOSEPAREN STATBLOCK  SEMICOLON  STATEMENTRPT
                | T_GET OPENPAREN VARIABLE  CLOSEPAREN SEMICOLON STATEMENTRPT
                | T_PUT OPENPAREN EXPR  CLOSEPAREN SEMICOLON STATEMENTRPT
                | T_RETURN OPENPAREN EXPR CLOSEPAREN SEMICOLON  STATEMENTRPT
                | EPSILON
VARDECLRPTSTDIS -> T_IDENTIFIER  ARRAYSIZERPT  SEMICOLON VARDECLRPTST |  IDNESTRPTIDDIIN ASSIGNOP  EXPR  SEMICOLON STATEMENTRPT
STATEMENTRPT    -> STATEMENT STATEMENTRPT | EPSILON
STATEMENT       -> T_IDENTIFIER IDNESTRPTIDDIIN ASSIGNOP  EXPR  SEMICOLON
                | T_IF OPENPAREN EXPR  CLOSEPAREN THEN  STATBLOCK ELSE  STATBLOCK  SEMICOLON
                | T_FOR  OPENPAREN TYPE  IDENTIFIER     ASSIGNOP  EXPR   SEMICOLON RELEXPR   SEMICOLON ASSIGNSTAT  CLOSEPAREN STATBLOCK  SEMICOLON
                | T_GET OPENPAREN VARIABLE  CLOSEPAREN SEMICOLON
                | T_PUT OPENPAREN EXPR  CLOSEPAREN SEMICOLON
                | T_RETURN OPENPAREN EXPR CLOSEPAREN SEMICOLON
EXPR            -> ARITHEXPR EXPRDISAMB
EXPRDISAMB      -> RELOP  ARITHEXPR  | EPSILON
ARITHEXPR       -> TERM ARITHEXPRRPT
ARITHEXPRRPT    -> ADDOP  TERM  ARITHEXPRRPT | EPSILON
RELEXPR         -> ARITHEXPR RELOP  ARITHEXPR
STATBLOCK       -> T_OPEN_CURLY STATEMENTRPT CLOSECURLY | STATEMENT | EPSILON
ASSIGNSTAT      -> VARIABLE ASSIGNOP  EXPR
VARIABLE        -> T_IDENTIFIER IDNESTRPTIDDIIN
IDNESTRPTIDDIFA ->  INDICERPT  IDNESTRPTIDDFD
                |  T_OPEN_PAREN APARAMS CLOSEPAREN
IDNESTRPTIDDFD  -> T_DOT IDENTIFIER IDNESTRPTIDDIFAU | EPSILON
IDNESTRPTIDDIFAU->  INDICERPT  IDNESTRPTIDDFD
                |  T_OPEN_PAREN APARAMS CLOSEPAREN
IDNESTRPTIDDIIN -> INDICERPT  INDICERPTDIS
INDICERPTDIS    -> T_DOT IDENTIFIER  INDICERPT  INDICERPTDIS | EPSILON
INDICERPT       -> T_OPEN_SQUARE ARITHEXPR  CLOSESQUARE INDICERPT | EPSILON
TERM            -> FACTOR TERMRPT
TERMRPT         -> MULTOP  FACTOR  TERMRPT | EPSILON
FACTOR          -> T_IDENTIFIER IDNESTRPTIDDIFA
                | T_INTEGER
                | T_FLOAT
                | T_OPEN_PAREN ARITHEXPR CLOSEPAREN
                | T_NOT FACTOR
                | SIGN FACTOR
APARAMS         -> EXPR  APARAMSTAILRPT | EPSILON
APARAMSTAILRPT  -> APARAMSTAIL APARAMSTAILRPT | EPSILON
